Emitter <Events>

Cross-platform, lightweight type-safe event emitter.

@example
type Events = {
  error: (error: Error) => void
  message: (from: string, content: string) => void
}

const emitter = new Emitter<Events>()
emitter.emit('error', 'x')  // will catch this type error
